# write a program function that takes a list and returns a new list that contains all the elements of the first list minus all the duplicates 

def unique_list(l):
  x = []
  for a in l:
    if a not in x:
      x.append(a)
  return x

print(unique_list([1,2,3,3,3,3,4,5]))
